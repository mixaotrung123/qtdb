
sudo apt update
sudo apt install subversion -y
svn --version

# Bước 1: Cài đặt Apache với module SVN
sudo apt update
sudo apt install apache2 libapache2-mod-svn

# Bước 2: Cấu hình Apache để sử dụng SVN
sudo nano /etc/apache2/mods-available/dav_svn.conf
# Thêm nội dung sau vào file dav_svn.conf:
# <Location /svn>
#    DAV svn
#    SVNParentPath /var/svn
#    AuthType Basic
#    AuthName "Subversion Repository"
#    AuthUserFile /etc/apache2/dav_svn.passwd
#    Require valid-user
# </Location>



# Tạo thư mục để chứa các kho lưu trữ SVN
sudo mkdir /var/svn
sudo svnadmin create /var/svn/myrepo

# Bước 3: Tạo người dùng cho SVN
# Tạo file mật khẩu và thêm người dùng đầu tiên
sudo htpasswd -cm /etc/apache2/dav_svn.passwd username

# Để thêm người dùng khác, sử dụng lệnh này (không sử dụng -c)
sudo htpasswd -m /etc/apache2/dav_svn.passwd another_username

# Bước 4: Kích hoạt module và khởi động lại Apache
sudo a2enmod dav
sudo a2enmod dav_svn
sudo systemctl restart apache2

Thay đổi quyền sở hữu thư mục
sudo chown -R www-data:www-data /var/svn/myrepo
Thay đổi quyền truy cập thư mục
sudo chmod -R 775 /var/lib/svn

# Truy cập kho lưu trữ SVN
# Bạn có thể truy cập kho lưu trữ SVN qua trình duyệt web hoặc client SVN khác bằng URL:
# http://your_server_ip/svn/myrepo

svn update /path/to/your/working/copy
svn status /path/to/your/working/copy
svn commit /path/to/your/working/copy -m "Your commit message"
